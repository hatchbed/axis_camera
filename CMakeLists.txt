cmake_minimum_required(VERSION 2.8.3)
project(axis_camera)

find_package(catkin REQUIRED
  COMPONENTS
  dynamic_reconfigure
  message_generation
)

# ROS message generation
add_message_files(FILES Axis.msg)
generate_messages()
generate_dynamic_reconfigure_options(cfg/PTZ.cfg)

catkin_package(
  CATKIN_DEPENDS dynamic_reconfigure message_runtime
)

install(PROGRAMS
  nodes/axis.py
  nodes/axis_ptz.py
  nodes/publish_axis_tf.py
  nodes/teleop.py
  nodes/teleop_speed_control.py
  DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})

install(DIRECTORY launch tests
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION})
